210  friend std::vector<T> operator*(const Matrix<T, Order>& matrix, const std::vector<T>& vec){
211  std::vector<T> result(matrix.numrows, T{0}); // Initialize result vector with zeros
212  if (!matrix.is_compressed()) {
213  // Uncompressed format
214  std::size_t i{0};
215  std::size_t j{0};
216 
217  // Traverse the matrix and compute dot product
218  for (const auto& [coords, value] : matrix.uncompressed_data) {
219  i = coords[0];
220  j = coords[1];
221  result[i] += value * vec[j];
222  }
223  } else {
224  // Compressed format
225  if constexpr(Order == StorageOrder::RowOrdering){
226  // Row ordering (CSR): traverse the matrix and perform classical row-times-vector algorithm
227  for (std::size_t i = 0; i < matrix.numrows; ++i) {
228  std::size_t row_start = matrix.compressed_inner[i];
229  std::size_t row_end = matrix.compressed_inner[i + 1];
230 
231  for (std::size_t k = row_start; k < row_end; ++k) {
232  std::size_t col_index = matrix.compressed_outer[k];
233  T value = matrix.compressed_data[k];
234  result[i] += value * vec[col_index]; // Compute dot product
235  }
236  }
237  }
238  else{
239  // Column ordering (CSC)
240  for (std::size_t j = 0; j < matrix.numcols; ++j) {
241  std::size_t col_start = matrix.compressed_inner[j];
242  std::size_t col_end = matrix.compressed_inner[j + 1];
243 
244  for (std::size_t k = col_start; k < col_end; ++k) {
245  std::size_t row_index = matrix.compressed_outer[k];
246  T value = matrix.compressed_data[k];
247  result[row_index] += value * vec[j]; // Compute linear combination
248  }
249  }
250  }
251  }
252  return result;
253  }
